CREATE TABLE USER_(
      USER_ID INT AUTO_INCREMENT PRIMARY KEY,
      FIRST_NAME VARCHAR(250),
      LAST_NAME VARCHAR(250),
      EMAIL VARCHAR(250),
      PH_NO BIGINT,
      GENDER VARCHAR(250)
);


CREATE TABLE ISSUES(
      USER_ID INT PRIMARY KEY,
      ISSUE VARCHAR(5000),
      CONSTRAINT FOREIGN KEY(USER_ID) REFERENCES USER_(USER_ID)
);


CREATE TABLE TICKETS(
      TICKET_ID INT AUTO_INCREMENT PRIMARY KEY,
      DESC_ VARCHAR(1000),
      PRICE INT      
);


CREATE TABLE TIMINGS(
      TICKET_ID INT,
      TIMING VARCHAR(250),
      CONSTRAINT FOREIGN KEY(TICKET_ID) REFERENCES TICKETS(TICKET_ID)
);


CREATE TABLE TICKET_BUYER(
      TICKET_BUYER_ID INT,
      TICKET_ID INT,
      BUYING_DATE DATE,
      DEPARTURE_DATE DATE,
      TIMING VARCHAR(250),
      URGENT VARCHAR(250),
      QUANTITY INT,
      TOTAL_PRICE INT,
      CONSTRAINT PRIMARY KEY(TICKET_BUYER_ID,TICKET_ID),
      CONSTRAINT FOREIGN KEY(TICKET_BUYER_ID) REFERENCES USER_(USER_ID),
      CONSTRAINT FOREIGN KEY(TICKET_ID) REFERENCES TICKETS(TICKET_ID)
);
